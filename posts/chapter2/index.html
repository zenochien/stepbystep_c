        private void pheptong()
        {
            int so1 = int.Parse(txt1.Text); //hạng số 1
            int so2 = int.Parse(txt2.Text); //hạng số 2
            int outcome = 0;
            //Xuất hiện kết quả
            outcome = so1 + so2;
            bieuhien.Text = $"{txt1.Text} + {txt2.Text}";
            ketqua.Text = outcome.ToString();
        }

        private void phantram()
        {
            int so1 = int.Parse(txt1.Text);
            int so2 = int.Parse(txt2.Text);
            int outcome = 0;

            outcome = so1 % so2;
            bieuhien.Text = $"{txt1.Text}+{txt2.Text}";
            ketqua.Text = outcome.ToString();
        }

        private void phepnhan()
        {
            int so1 = int.Parse(txt1.Text);
            int so2 = int.Parse(txt2.Text);
            int outcome = 0;

            outcome = so1 * so2;
            bieuhien.Text = $"{txt1.Text}+{txt2.Text}";
            ketqua.Text = outcome.ToString();
        }

        private void phepchia()
        {
            int so1 = int.Parse(txt1.Text);
            int so2 = int.Parse(txt2.Text);
            int outcome = 0;

            outcome = so1 / so2;
            bieuhien.Text = $"{txt1.Text}+{txt2.Text}";
            ketqua.Text = outcome.ToString();
        }

        private void pheptru()
        {
            int so1 = int.Parse(txt1.Text);
            int so2 = int.Parse(txt2.Text);
            int outcome = 0;

            outcome = so1 - so2;
            bieuhien.Text = $"{txt1.Text} + {txt2.Text}";
            ketqua.Text = outcome.ToString();
        }

        private void tinhClick(object sender, RoutedEventArgs e)
        {
            try
            {
                if ((bool)tong.IsChecked)
                {
                    pheptong();
                }
                else if ((bool)tru.IsChecked)
                {
                    pheptru();
                }
                else if ((bool)chia.IsChecked)
                {
                    phepchia();
                }
                else if ((bool)nhan.IsChecked)
                {
                    phepnhan();
                }
                else if ((bool)phan.IsChecked)
                {
                    phantram();
                }
            }
            catch (Exception caught)
            {
                bieuhien.Text = "";
                ketqua.Text = caught.Message;
            }
        }

Kiểu dữ liệu

private void kieudulieu_SelectionChanged(object sender, SelectionChangedEventArgs e)
        {
                ListBoxItem selectedType = (kieudulieu.SelectedItem as ListBoxItem);
                switch (selectedType.Content.ToString())
                {
                    case "int": 
                        showIntValue();
                        break;
                    case "long": 
                        showLongValue();
                        break;
                    case "float": 
                        showFloatValue();
                        break;
                    case "double":
                        showDoubleValue();
                        break;
                    case "decimal":
                        showDecimalValue();
                        break;
                    case "string":
                        showStringValue();
                        break;
                    case "char":
                        showCharValue();
                        break;
                    case "bool":
                        showBoolValue();
                        break;
                }
            }

            private void showIntValue()
            {
                int intVar;
                intVar = 42;
                value.Text = intVar.ToString();
            }

            private void showLongValue()
            {
                long longVar;
                longVar = 42L;
                value.Text = longVar.ToString();
            }

            private void showFloatValue()
            {
                float floatVar;
                floatVar = 0.42F;
                value.Text = floatVar.ToString();
            }

            private void showDoubleValue()
            {
                double doubleVar;
                doubleVar = 0.42;
                value.Text = doubleVar.ToString();
            }

            private void showDecimalValue()
            {
                decimal decimalVar;
                decimalVar = 0.42M;
                value.Text = decimalVar.ToString();
            }

            private void showStringValue()
            {
                string stringVar;
                stringVar = "forty two";
                value.Text = stringVar; 
            }

            private void showCharValue()
            {
                char charVar;
                charVar = 'x';
                value.Text = charVar.ToString();
            }

            private void showBoolValue()
            {
                bool boolVar;
                boolVar = false;
                value.Text = boolVar.ToString();
            }
        }
